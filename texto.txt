const productos = [

nombre:"Fat Burner Supplement",

}

]

function buscar(){

let input = document.getElementById("search").value.toLowerCase()

let resultados = productos.filter(p => p.keyword.includes(input))

let html=""

if(resultados.length===0){
html="<p>No supplements found</p>"
}

resultados.forEach(p=>{
html += `
<div class="product">
<h3>${p.nombre}</h3>
<a href="${p.link}" target="_blank">View on TikTok Shop</a>
</div>
`
})

document.getElementById("resultados").innerHTML = html

}
